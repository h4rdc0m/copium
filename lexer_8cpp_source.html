46Token Lexer::scanToken() {
47 skipWhitespace();
48
49 start = current;
50
51 if (isAtEnd()) {
52 return Token(TokenType::END_OF_FILE, "", line, column);
53 }
54
55 // Store the starting column position for this token
56 int startColumn = column;
57
58 char c = advance();
59
60 // Handle identifiers
61 if (std::isalpha(c) || c == '_') {
62 scanIdentifier();
63
64 std::string text = source.substr(start, current - start);
65 TokenType type = TokenType::IDENTIFIER;
66
67 // Check if the identifier is a keyword
68 auto it = keywords.find(text);
69 if (it != keywords.end()) {
70 type = it->second;
71 }
72
73 return Token(type, text, line, startColumn);
74 }
75
76 // Handle numbers
77 if (std::isdigit(c)) {
78 scanNumber();
79 std::string text = source.substr(start, current - start);
80 TokenType type =
81 text.find('.') != std::string::npos ? TokenType::FLOAT_LITERAL : TokenType::INTEGER_LITERAL;
82 return Token(type, text, line, startColumn); // Use startColumn instead of column
83 }
84
85 // Handle single-character tokens
86 switch (c) {
87 case '(':
88 return Token(TokenType::LEFT_PAREN, "(", line, startColumn);
89 case ')':
90 return Token(TokenType::RIGHT_PAREN, ")", line, startColumn);
91 case '{':
92 return Token(TokenType::LEFT_BRACE, "{", line, startColumn);
93 case '}':
94 return Token(TokenType::RIGHT_BRACE, "}", line, startColumn);
95 case '[':
96 return Token(TokenType::LEFT_BRACKET, "[", line, startColumn);
97 case ']':
98 return Token(TokenType::RIGHT_BRACKET, "]", line, startColumn);
99 case '.':
100 return Token(TokenType::DOT, ".", line, startColumn);
101 case ',':
102 return Token(TokenType::COMMA, ",", line, startColumn);
103 case ';':
104 return Token(TokenType::SEMICOLON, ";", line, startColumn);
105 case '%':
106 return Token(TokenType::PERCENT, "%", line, startColumn);
107
108 // Handle two-character tokens
109 case '!':
110 if (match('=')) {
111 return Token(TokenType::NOT_EQUAL, "!=", line, startColumn);
112 }
113 return Token(TokenType::NOT, "!", line, startColumn);
114 case '=':
115 if (match('=')) {
116 return Token(TokenType::EQUAL_EQUAL, "==", line, startColumn);
117 }
118 return Token(TokenType::EQUAL, "=", line, startColumn);
119 case '<':
120 if (match('=')) {
121 return Token(TokenType::LESS_EQUAL, "<=", line, startColumn);
122 }
123 return Token(TokenType::LESS, "<", line, startColumn);
124 case '>':
125 if (match('=')) {
126 return Token(TokenType::GREATER_EQUAL, ">=", line, startColumn);
127 }
128 return Token(TokenType::GREATER, ">", line, startColumn);
129 case '+':
130 if (match('=')) {
131 return Token(TokenType::PLUS_EQUAL, "+=", line, startColumn);
132 }
133 return Token(TokenType::PLUS, "+", line, startColumn);
134 case '-':
135 if (match('>')) {
136 return Token(TokenType::ARROW, "->", line, startColumn);
137 }
138 if (match('=')) {
139 return Token(TokenType::MINUS_EQUAL, "-=", line, startColumn);
140 }
141 return Token(TokenType::MINUS, "-", line, startColumn);
142 case '*':
143 if (match('=')) {
144 return Token(TokenType::STAR_EQUAL, "*=", line, startColumn);
145 }
146 return Token(TokenType::STAR, "*", line, startColumn);
147 case '/':
148 if (match('/')) {
149 // A comment goes until the end of the line
150 while (peek() != '\n' && !isAtEnd()) {
151 advance();
152 }
153 return scanToken(); // Skip the comment and get the next token
154 }
155 if (match('*')) {
156 // Handle multi-line comment
157 skipComment();
158 return scanToken(); // Skip the comment and get the next token
159 }
160 if (match('=')) {
161 return Token(TokenType::SLASH_EQUAL, "/=", line, startColumn);
162 }
163 return Token(TokenType::SLASH, "/", line, startColumn);
164 case '&':
165 if (match('&')) {
166 return Token(TokenType::AND, "&&", line, startColumn);
167 }
168 errorToken("Unexpected character: '&', did you mean '&&'?");
169 return Token(TokenType::ERROR, "&", line, startColumn);
170 case '|':
171 if (match('|')) {
172 return Token(TokenType::OR, "||", line, startColumn);
173 }
174 errorToken("Unexpected character: '|', did you mean '||'?");
175 return Token(TokenType::ERROR, "|", line, startColumn);
176 case ':':
177 if (match(':')) {
178 return Token(TokenType::COLON_COLON, "::", line, startColumn);
179 }
180 return Token(TokenType::COLON, ":", line, startColumn);
181 case '"':
182 scanString();
183 if (isAtEnd() || peek() != '"') {
184 errorToken("Unterminated string.");
185 return Token(TokenType::ERROR, source.substr(start, current - start), line, startColumn);
186 }
187 // Consume the closing quote
188 advance();
189 return Token(TokenType::STRING_LITERAL,
190 source.substr(start + 1, current - start - 2),
191 line,
192 startColumn);
193 default:
194 // Handle unexpected characters
195 errorToken("Unexpected character: '" + std::string(1, c) + "'");
196 return Token(TokenType::ERROR, std::string(1, c), line, startColumn);
197 }

288void Lexer::skipComment() {
289 int nesting = 1;
290 while (nesting > 0 && !isAtEnd()) {
291 if (peek() == '*' && peekNext() == '/') {
292 advance(); // Consume '*'
293 advance(); // Consume '/'
294 nesting--;
295 } else if (peek() == '/' && peekNext() == '*') {
296 advance(); // Consume '/'
297 advance(); // Consume '*'
298 nesting++;
299 } else if (peek() == '\n') {
300 line++;
301 column = 1;
302 advance();
303 } else {
304 advance();
305 }
306 }
307
308 if (nesting > 0) {
309 errorToken("Unterminated multi-line comment.");
310 }
